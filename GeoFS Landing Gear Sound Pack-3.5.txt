// ==UserScript==
// @name         GeoFS Landing Gear Sound Pack
// @namespace    http://tampermonkey.net/
// @version      3.5
// @description  Gear sound with fenced AudioContext, mute/pause integration, quieter volume, cockpit/jumpseat/wing2 support, and seamless click-free looping.
// @match        https://www.geo-fs.com/*
// @match        https://beta.geo-fs.com/geofs.php
// @match        https://geo-fs.com/*
// @match        https://www.geo-fs.com/geofs.php*
// @match        https://geo-fs.com/geofs.php*
// @match http://*/geofs.php*
// @match https://*/geofs.php*
// @grant        none
// ==/UserScript==

(function() {
    "use strict";

    console.log("[GEAR SOUND] Userscript loaded into page");

    // Create hidden iframe to obtain a clean AudioContext
    const iframe = document.createElement("iframe");
    iframe.style.display = "none";
    document.body.appendChild(iframe);

    const CleanAudioContext =
        iframe.contentWindow.AudioContext ||
        iframe.contentWindow.webkitAudioContext;

    // Wait for GeoFS to fully initialize
    const waitForGeoFS = setInterval(() => {
        if (
            window.geofs &&
            geofs.camera &&
            typeof geofs.camera.currentView === "string" &&
            geofs.animation &&
            geofs.animation.values &&
            geofs.aircraft &&
            geofs.aircraft.instance
        ) {
            clearInterval(waitForGeoFS);
            console.log("[GEAR SOUND] GeoFS fully ready — initializing");
            initGearSoundSystem();
        }
    }, 300);

    function initGearSoundSystem() {
        console.log("[GEAR SOUND] initGearSoundSystem() called");

        const gearSoundURL =
            "https://raw.githubusercontent.com/ChristianPilotAlex003/GeoFS-B777-Sound-Pack-Official/main/Gear%20Down%20(3).wav";

        const allowedViews = ["cockpit", "jumpseat", "Wing 2", "wing2", "left wing", "right wing", "Left Wing", "Right Wing","Left wing", "Right wing"];

        // Crossfade length (seconds) applied to the loop seam to avoid clicks/cracks
        const LOOP_CROSSFADE_SEC = 0.02;

        // Fenced AudioContext
        const ctx = new CleanAudioContext();
        let buffer = null;
        let source = null;
        let gain = null;

        // Expose for debugging
        window._GEAR_ctx = ctx;
        window._GEAR_buffer = () => buffer;
        window._GEAR_source = () => source;
        window._GEAR_gain = () => gain;

        // Auto-resume
        document.addEventListener("click", () => ctx.resume());
        window.addEventListener("focus", () => ctx.resume());

        // -----------------------------------------------------------
        // Applies a short crossfade between the end and start of the
        // buffer so looping doesn't produce an audible click/crack.
        // fadeSec: length of the crossfade window in seconds
        // -----------------------------------------------------------
        function createSeamlessLoopBuffer(srcBuffer, fadeSec) {
            const sampleRate = srcBuffer.sampleRate;
            const fadeSamples = Math.floor(fadeSec * sampleRate);
            const length = srcBuffer.length;

            if (fadeSamples * 2 >= length) return srcBuffer; // too short, skip

            const outBuffer = ctx.createBuffer(
                srcBuffer.numberOfChannels,
                length,
                sampleRate
            );

            for (let ch = 0; ch < srcBuffer.numberOfChannels; ch++) {
                const inData = srcBuffer.getChannelData(ch);
                const outData = outBuffer.getChannelData(ch);

                // Copy original data
                for (let i = 0; i < length; i++) {
                    outData[i] = inData[i];
                }

                // Crossfade: blend the start of the buffer into the end,
                // so the tail eases toward the head's values before looping
                for (let i = 0; i < fadeSamples; i++) {
                    const fadeOut = 1 - (i / fadeSamples); // 1 -> 0 across the tail
                    const fadeIn = i / fadeSamples;        // 0 -> 1 across the head

                    const tailIdx = length - fadeSamples + i;
                    const headSample = inData[i];
                    const tailSample = inData[tailIdx];

                    // Blend tail toward head value
                    outData[tailIdx] = tailSample * fadeOut + headSample * fadeIn;
                }
            }

            return outBuffer;
        }

        // Load WAV
        fetch(gearSoundURL)
            .then((r) => r.arrayBuffer())
            .then((b) => ctx.decodeAudioData(b))
            .then((decoded) => {
                buffer = createSeamlessLoopBuffer(decoded, LOOP_CROSSFADE_SEC);
                console.log("[GEAR SOUND] WAV decoded and loop-smoothed");
            })
            .catch((err) =>
                console.error("[GEAR SOUND] WAV decode error:", err)
            );

        function startGearSound() {
            if (!buffer) return;
            if (source) return;

            source = ctx.createBufferSource();
            gain = ctx.createGain();

            source.buffer = buffer;
            source.loop = true;

            source.connect(gain);
            gain.connect(ctx.destination);

            // Quieter volume (0.60)
            gain.gain.setValueAtTime(0, ctx.currentTime);
            gain.gain.linearRampToValueAtTime(
                0.60,
                ctx.currentTime + 0.25
            );

            source.start(0);
            console.log("[GEAR SOUND] START");
        }

        function stopGearSound() {
            if (!source) return;

            gain.gain.cancelScheduledValues(ctx.currentTime);
            gain.gain.setValueAtTime(gain.gain.value, ctx.currentTime);
            gain.gain.linearRampToValueAtTime(
                0,
                ctx.currentTime + 0.2
            );

            setTimeout(() => {
                if (source) {
                    source.stop(0);
                    source.disconnect();
                }
                if (gain) gain.disconnect();

                source = null;
                gain = null;

                console.log("[GEAR SOUND] STOP");
            }, 250);
        }

        // Main loop
        setInterval(() => {
            const gearPos = geofs.animation.values.gearPosition;
            const view = geofs.camera.currentView;

            // Respect your global mute/pause system
            const globallyMuted =
                window._GE90_persistentlyMuted ||
                window._GE90_userMuted ||
                window._GE90_paused;

            const inAllowedView = allowedViews.includes(view);
            const gearMoving = gearPos > 0.01 && gearPos < 0.99;

            // If muted or paused → force stop
            if (globallyMuted) {
                stopGearSound();
                return;
            }

            if (gearMoving && inAllowedView) {
                startGearSound();
            } else {
                stopGearSound();
            }
        }, 120);
    }
})();
